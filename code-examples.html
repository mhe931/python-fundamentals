# -----------------------------------------------------------------------------
# Python Fundamentals: Hands-On Code Examples for Your Workshop
# -----------------------------------------------------------------------------

# This script provides simple, runnable examples for the core Python concepts
# covered in your presentation. Encourage your students to copy and paste
# these blocks into their Python interpreter or a separate file to experiment
# with the code.

# -----------------------------------------------------------------------------
# SECTION 1: Data Types and Variables
# -----------------------------------------------------------------------------
# Variables are like labels for values. Python figures out the type automatically.
print("--- SECTION 1: Data Types and Variables ---")

# A variable storing an integer (whole number)
student_age = 22
print(f"Student age: {student_age} (Type: {type(student_age)})")

# A variable storing a float (decimal number)
gpa = 3.8
print(f"GPA: {gpa} (Type: {type(gpa)})")

# A variable storing a string (text)
student_name = "Alice"
print(f"Student name: '{student_name}' (Type: {type(student_name)})")

# A variable storing a boolean (True or False)
is_enrolled = True
print(f"Is enrolled: {is_enrolled} (Type: {type(is_enrolled)})")

# -----------------------------------------------------------------------------
# SECTION 2: Operators and Expressions
# -----------------------------------------------------------------------------
# Operators perform actions on values.
print("\n--- SECTION 2: Operators and Expressions ---")

# Arithmetic Operators
a = 10
b = 5
print(f"Arithmetic: {a} + {b} = {a + b}")
print(f"Arithmetic: {a} * {b} = {a * b}")

# Comparison Operators (return booleans)
c = 7
d = 12
print(f"Comparison: Is {c} greater than {d}? {c > d}")
print(f"Comparison: Is {c} not equal to {d}? {c != d}")

# Logical Operators
is_python_fun = True
has_experience = False
print(f"Logical: Is Python fun AND do they have experience? {is_python_fun and has_experience}")
print(f"Logical: Is Python fun OR do they have experience? {is_python_fun or has_experience}")

# -----------------------------------------------------------------------------
# SECTION 3: Control Flow
# -----------------------------------------------------------------------------
print("\n--- SECTION 3: Control Flow ---")

# Conditional Statements (if/else)
age = 17
if age >= 18:
    print("You are eligible to vote.")
else:
    print("You are not old enough to vote.")

# A for loop to iterate through a list
print("\nFor Loop Example:")
fruits = ["apple", "banana", "cherry"]
for fruit in fruits:
    print(f"I have a {fruit}.")

# A while loop that runs until a condition is met
print("\nWhile Loop Example:")
count = 0
while count < 3:
    print(f"Count is {count}")
    count += 1

# -----------------------------------------------------------------------------
# SECTION 4: Data Structures
# -----------------------------------------------------------------------------
print("\n--- SECTION 4: Data Structures ---")

# Lists: Ordered and changeable
my_list = [10, "hello", 3.14]
my_list.append("world")  # Adding an item
print(f"List: {my_list}")
print(f"List element at index 1: {my_list[1]}")

# Dictionaries: Key-value pairs
my_dict = {"name": "Charlie", "age": 30, "city": "Vaasa"}
print(f"Dictionary: {my_dict}")
print(f"Charlie's age is: {my_dict['age']}")

# Tuples: Ordered and unchangeable
my_tuple = (1, 2, 3)
print(f"Tuple: {my_tuple}")
# The following line would cause an error: my_tuple.append(4)

# Sets: Unordered and unique items
my_set = {1, 2, 3, 3, 4}
print(f"Set with duplicates removed: {my_set}")
